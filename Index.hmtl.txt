import React, { useState } from "react";
import { Card, CardContent } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Textarea } from "@/components/ui/textarea";

export default function BearArmsMilitia() {
  const [formData, setFormData] = useState({ name: "", email: "", message: "" });
  const [formStatus, setFormStatus] = useState("");

  const handleChange = (e) => {
    const { name, value } = e.target;
    setFormData({ ...formData, [name]: value });
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    setFormStatus("Sending...");

    try {
      const response = await fetch("/api/contact", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(formData),
      });

      if (response.ok) {
        setFormStatus("Message sent successfully!");
        setFormData({ name: "", email: "", message: "" });
      } else {
        setFormStatus("Failed to send message. Please try again.");
      }
    } catch (error) {
      setFormStatus("An error occurred. Please try again.");
    }
  };

  return (
    <div className="min-h-screen bg-black text-yellow-500">
      {/* Header */}
      <header className="bg-black text-yellow-500 p-4 shadow-lg sticky top-0 z-50">
        <div className="container mx-auto flex justify-between items-center">
          <h1 className="text-2xl font-bold">Bear Arms Militia</h1>
          <nav className="space-x-4">
            <a href="#shop" className="hover:text-yellow-300">Shop</a>
            <a href="#contact" className="hover:text-yellow-300">Contact</a>
            <a href="#faq" className="hover:text-yellow-300">FAQ</a>
          </nav>
        </div>
      </header>

      {/* Main Content */}
      <main>
        {/* Home Section */}
        <section className="bg-cover bg-center h-96 flex items-center justify-center" style={{ backgroundImage: 'url("/path-to-banner-image.jpg")' }}>
          <div className="text-center text-yellow-500 bg-black bg-opacity-75 p-8 rounded animate-fade-in">
            <h2 className="text-4xl font-bold mb-4">Welcome to Bear Arms Militia</h2>
            <p className="mb-6">Discover premium merchandise, exclusive art, and more.</p>
            <a href="#shop" className="bg-yellow-500 text-black px-6 py-3 rounded shadow hover:bg-yellow-300">Shop Now</a>
          </div>
        </section>

        {/* Shop Section */}
        <section id="shop" className="py-16">
          <div className="container mx-auto">
            <h2 className="text-3xl font-bold text-center mb-8 text-yellow-500">Shop</h2>
            <div className="grid grid-cols-1 md:grid-cols-3 gap-8">
              {[1, 2, 3].map((product) => (
                <Card key={product} className="shadow-md bg-black border border-yellow-500 transform transition duration-500 hover:scale-105">
                  <CardContent className="text-center">
                    <img src={`/path-to-product-${product}.jpg`} alt="Product" className="mb-4 w-full h-40 object-cover rounded" />
                    <h3 className="font-semibold text-lg text-yellow-500">Product Name</h3>
                    <p className="text-yellow-300">$99.99</p>
                    <Button className="mt-4 w-full bg-yellow-500 text-black hover:bg-yellow-300">Add to Cart</Button>
                  </CardContent>
                </Card>
              ))}
            </div>
          </div>
        </section>

        {/* Contact Section */}
        <section id="contact" className="bg-black py-16">
          <div className="container mx-auto">
            <h2 className="text-3xl font-bold text-center mb-8 text-yellow-500">Contact Us</h2>
            <form onSubmit={handleSubmit} className="max-w-lg mx-auto bg-black p-8 rounded shadow border border-yellow-500">
              <Input
                name="name"
                placeholder="Your Name"
                value={formData.name}
                onChange={handleChange}
                className="mb-4 bg-black text-yellow-500 border border-yellow-300"
              />
              <Input
                type="email"
                name="email"
                placeholder="Your Email"
                value={formData.email}
                onChange={handleChange}
                className="mb-4 bg-black text-yellow-500 border border-yellow-300"
              />
              <Textarea
                name="message"
                placeholder="Your Message"
                value={formData.message}
                onChange={handleChange}
                className="mb-4 bg-black text-yellow-500 border border-yellow-300"
              />
              <Button type="submit" className="w-full bg-yellow-500 text-black hover:bg-yellow-300">Send</Button>
              {formStatus && <p className="mt-4 text-center text-yellow-300">{formStatus}</p>}
            </form>
          </div>
        </section>

        {/* FAQ Section */}
        <section id="faq" className="py-16">
          <div className="container mx-auto">
            <h2 className="text-3xl font-bold text-center mb-8 text-yellow-500">FAQ</h2>
            <div className="space-y-4">
              <div className="bg-black p-4 rounded shadow border border-yellow-500">
                <h3 className="font-semibold text-lg text-yellow-500">What is Bear Arms Militia?</h3>
                <p className="text-yellow-300>Something the big Mexican asked me to while pointing a gun to my head.</p>
              </div>
              <div className="bg-black p-4 rounded shadow border border-yellow-500">
                <h3 className="font-semibold text-lg text-yellow-500">How can I track my order?</h3>
                <p className="text-yellow-300">You will receive a tracking link via email after your order is shipped.</p>
              </div>
            </div>
          </div>
        </section>
      </main>

      {/* Footer */}
      <footer className="bg-black text-yellow-500 py-8">
        <div className="container mx-auto text-center space-y-4">
          <p>&copy; 2025 Bear Arms Militia. All rights reserved.</p>
          <div className="flex justify-center space-x-6">
            <a href="https://www.instagram.com/beararmsmilitia/" target="_blank" className="hover:text-yellow-300">Instagram</a>
            <a href="https://www.tiktok.com/@bear.arms.militia" target="_blank" className="hover:text-yellow-300">TikTok</a>
            <a href="https://www.youtube.com/@BArmsMilitia" target="_blank" className="hover:text-yellow-300">YouTube</a>
            <a href="https://www.patreon.com/BearArmsMilitia" target="_blank" className="hover:text-yellow-300">Patreon</a>
            <a href="https://x.com/BArmsMilitia" target="_blank" className="hover:text-yellow-300">Twitter</a>
          </div>
          <div className="mt-4">
            <p className="text-sm">Follow us for updates, exclusive content, and promotions!</p>
          </div>
        </div>
      </footer>
    </div>
  );
}
